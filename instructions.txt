#   500 "the current context does not logits computation. skipping"

from llama_cpp import Llama

# Initialize the model strictly for embedding tasks
embd = Llama(
    model_path="./nomic-embed-text-v1.5.Q8_0.gguf", 
    embedding=True, 
    n_ctx=8192
)

# Nomic requires a search prefix prefix depending on the task
# Options: 'search_query: ' or 'search_document: '
text = "search_query: What is the optimal pipeline structure for autonomous AI agents?"
output = embd.embed(text)

print("Embedding vector length:", len(output))
print("Vector preview:", output[:5])
